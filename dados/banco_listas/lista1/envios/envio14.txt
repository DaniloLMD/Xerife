#include <iostream>
#include <vector>

using namespace std;

#define QTD_NOTAS 6
#define MAX_SAQUE 5001

vector<int> pesos = {2, 5, 10, 20, 50, 100};
vector<vector<int>> dp(MAX_SAQUE, vector<int>(QTD_NOTAS, -1));
vector<int> notas(QTD_NOTAS);

int banco(int saque, int nota_atual=0){
    if(saque == 0) return 1;
    if(saque < 0 /*|| nota_atual >= QTD_NOTASpesos*/) return 0;

    if(dp[saque][nota_atual] != -1) return dp[saque][nota_atual];

    int tot = 0;
    
    if(nota_atual == QTD_NOTAS-1){
        if((double) saque/pesos[nota_atual] == (int) saque/pesos[nota_atual]) tot++;
        return dp[saque][nota_atual] = tot;
    }
    for(int i = 0; i <= notas[nota_atual]; i++){
        if(saque < i * pesos[nota_atual]) break; 
        tot += banco(saque - i * pesos[nota_atual], nota_atual+1);
    }

    dp[saque][nota_atual] = tot;
    return tot;
}
